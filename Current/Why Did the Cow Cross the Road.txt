#include <bits/stdc++.h>
using namespace std;
struct cow{
    int id;
    int pos;
};
struct info{
    int cross;
    int pos;
};
int main(){
    int length;
    cin >> length;
    cow input[length];
    for (int i = 0; i < length; i++){
        cin >> input[i].id;
        cin >> input[i].pos;
        input[i].pos++;
    }
    map<int, info> m;
    for (int i = 0; i < length; i++){
        if (m[input[i].id].pos == 0){
            m[input[i].id].pos = input[i].pos;
        }
        else{
            if (m[input[i].id].pos != input[i].pos){
                m[input[i].id].cross++;
                m[input[i].id].pos = input[i].pos;
            }
        }
    }
    
}
