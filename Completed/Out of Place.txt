#include <bits/stdc++.h>
using namespace std;
int main(){
    int length;
    cin >> length;
    int arr[length];
    int sortarr[length];
    for (int i = 0; i < length; i++){
        cin >> arr[i];
        sortarr[i] = arr[i];
    }
    sort(arr, arr + length);
    int ans = 0;
    for (int i = 0; i < length; i++){
        if (sortarr[i] != arr[i])
            ans++;
    }
    cout << ans - 1;
 
}
