#include <iostream>
using namespace std;
struct cow{
    int before;
    int after;
};
int main(){
    cow input[4];
    for (int i = 0; i < 4; i++){
        cin >> input[i].before;
        cin >> input[i].after;
    }
    int beforetotal = 0, aftertotal = 0;
    for (int i = 0; i < 4; i++){
        beforetotal += input[i].before;
        aftertotal += input[i].after;
    }
    input[0].before += aftertotal - beforetotal;
    cout << input[0].before - input[0].after << " ";
    input[1].before += input[0].before - input[0].after;
    cout << input[1].before - input[1].after << " ";
    input[2].before += input[1].before - input[1].after;
    cout << input[2].before - input[2].after << " ";


}
